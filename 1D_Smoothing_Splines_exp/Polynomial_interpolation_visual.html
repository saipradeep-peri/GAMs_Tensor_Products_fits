y_func <- function(x){
  return(1/(1+25*x^2))
}

#The polynomial interpolation function calculated above
polynomial_interpolant <- function(x, y){
  return(1 - 16.8552*x^2 + 123.3597*x^4 - 381.4338*x^6 + 494.9095*x^8 - 220.9417*x^10)
}
